471  point_cloud_graph&
472  operator= (const point_cloud_graph& x)
473  {
474  if (&x != this)
475  {
476  /* The motivation behind this `reset` invokation is as follows. The
477  * `operator=` function of the base class (`vec_adj_list_impl`) will
478  * use the standard `add_vertex` to append each vertex of the source
479  * graph one after another. Apparently, this is not the fastest way
480  * to copy the underlying point cloud. Resetting the point cloud
481  * pointer effectively makes `added_vertex` a no-op, so the time is
482  * not wasted copying points one by one. Rather in the next line we
483  * make a complete copy with one call. */
484  m_point_cloud.reset ();
485  Base::operator= (x);
486  m_point_cloud.reset (new point_cloud_type (*x.m_point_cloud));
487  }
488  return (*this);
489  }

501  /* A note on the implementation of vertex addition/removal.
502  *
503  * In BGL vertices are added and removed to a graph via non-member
504  * functions `add_vertex` and `remove_vertex`. The implementation that is
505  * provided for `vec_adj_list_impl` (from which this class actually
506  * derives) does everything that is needed, except to adding/removing
507  * points from internal point cloud (for obvious reasons). In order to
508  * augment the behavior we could have specialized these functions for
509  * `point_cloud_graph` class, however that would involve copy-pasting of
510  * quite some code. Luckily, these functions have some sort of hooks
511  * which are member functions of graph object and got called in the end,
512  * namely `added_vertex` and `removing_vertex`. (In fact, they are
513  * designated to support some `named_graph` extension, but who cares.)
514  * Therefore these two hooks are implemented in `point_cloud_graph` to
515  * perform the desired point cloud maintenance. */
516 
517  void
518  added_vertex (vertex_descriptor vertex)
519  {
520  if (m_point_cloud)
521  m_point_cloud->push_back (point_type ());
522  }
523 
524  void
525  removing_vertex (vertex_descriptor vertex)
526  {
527  if (m_point_cloud)
528  m_point_cloud->erase (m_point_cloud->begin () + vertex);
529  }
