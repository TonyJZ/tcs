222  std::function<float (float)> getRangeScalingForEdges ()
223  {
224  float max = -std::numeric_limits<float>::infinity();
225  float min = std::numeric_limits<float>::infinity();
226  for (const auto& s : as_range (boost::edges (graph_)))
227  {
228  auto v = boost::get (boost::edge_weight_t (), graph_, s);
229  //v = std::log (v);
230  if (v > max) max = v;
231  if (v < min) min = v;
232  }
233  if (max != min)
234  //return [min, max] (float v) { return (std::log (v) - min) / (max - min); };
235  return [min, max] (float v) { return (v - min) / (max - min); };
236  else
237  return [] (float v) { return 1.0; };
238  }

240  std::function<float (float)> getRangeScalingForVector (const Eigen::VectorXf& values)
241  {
242  float max = values.maxCoeff ();
243  float min = values.minCoeff ();
244  float range = max - min;
245  return [min, range] (float v) { return (v - min) / range; };
246  }
